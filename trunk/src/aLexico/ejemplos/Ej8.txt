@ Programa de ejemplo
program: ejemplo{
      vars-consts{
var integer cantidad;
var float euros;
var float centimos;
var float pesetas;
const float razon := 166.386;
}
instructions{
in(euros);;
cantidad = (int)euros;
centimos = euros*cantidad;
out(cantidad);

swap2();
pesetas = euros / razon;
out (pesetas);
}
}
